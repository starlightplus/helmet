<script setup>
import { ref, computed, onMounted, onUnmounted } from 'vue';
import { use } from 'echarts/core';
import { CanvasRenderer } from 'echarts/renderers';
import { LineChart } from 'echarts/charts';
import { GridComponent, TooltipComponent } from 'echarts/components';
import VChart from 'vue-echarts';

// 注册 ECharts 模块

const timestamp = ref(new Date().toISOString());

const currentTemp = ref(24.5);

const cards = computed(() => [
  { label: 'Ambient Temperature', value: currentTemp.value, unit: '°C', type: 'temp', color: 'cyan', max: 50 },

const chartOption = computed(() => ({
  backgroundColor: 'transparent',
  // 1. 开启全局平滑动画
  animationDuration: 1000,
  animationDurationUpdate: 1000,
  animationEasingUpdate: 'cubicOut',
  tooltip: {
    trigger: 'axis',
    backgroundColor: 'rgba(0,0,0,0.8)',

  grid: { top: '10%', left: '3%', right: '3%', bottom: '5%', containLabel: true },

    type: 'category',
    boundaryGap: false, // 2. 消除边缘间隙，让平移更自然
    data: historyData.value.map(d => d.time),
    axisLine: { show: false },
    axisLabel: { color: 'rgba(255,255,255,0.2)', fontSize: 10, fontFamily: 'JetBrains Mono' }

  yAxis: {

    axisLabel: { color: 'rgba(255,255,255,0.2)', fontSize: 10, fontFamily: 'JetBrains Mono' }
  },

      type: 'line',

      }

      name: 'HUM',
      type: 'line',
      smooth: 0.4,
      showSymbol: false,

      lineStyle: {

        width: 2,
        shadowBlur: 10,
        shadowColor: 'rgba(255,0,85,0.5)' 
      },
      areaStyle: {
        color: {
          type: 'linear', x: 0, y: 0, x2: 0, y2: 1,
          colorStops: [{ offset: 0, color: 'rgba(255,0,85,0.1)' }, { offset: 1, color: 'rgba(255,0,85,0)' }]

      }
    }
  ]

// --- 模拟实时数据流 ---

onMounted(() => {

    time: `${i}:00`,

    currentHum.value = nextHum;

      ...historyData.value.slice(-19), 
      { time: timeStr, temp: nextTemp, hum: nextHum }
    ];
  }, 3000);
});

onUnmounted(() => clearInterval(timer));
</script>

### 为什么这个代码能解决你的问题？

1. 
2. **yAxis.scale: true**：这是最关键的一点。它让 Y 轴紧贴你的数据波动范围（比如 24.1 到 24.8），而不是死板地显示 0 到 100。这会极大地放大波动的视觉效果。
3. **smooth: 0.4**：它将折线变成了平滑的曲线。当数据点向左平移时，曲线会像液体一样流动。
4. **animationDurationUpdate: 1000**：这设定了数据变化时的“补间动画”时长。如果这个值太小，你会看到线条在跳动；设定为 1000ms 后，你会看到线条在缓慢、优雅地滑向新位置。
5. **boundaryGap: false**：这让线条一直延伸到图表的最边缘，消除左右两侧的空白，增强了数据“流过”屏幕的沉浸感。
