<script setup lang="ts">
import { ref, reactive, computed, onMounted, watch } from 'vue';
import { 
  User, 
  Calendar, 
  Scale, 
  Phone, 
  Heart, 
  Edit3, 
  Check, 
  X, 
  Upload, 
  AlertCircle, 
  ShieldAlert, 
  Activity, 
  UserCircle, 
  Sparkles, 
  Weight, 
  Save, 
  FileText,
  BadgeAlert,
  Smartphone,
  Flame,
  CheckCircle2,
  Trash2
} from 'lucide-vue-next';

// Built-in presets for gorgeous gradient avatars
const gradientPresets = [
  { name: '晨曦紫罗兰', bg: 'linear-gradient(135deg, #a78bfa 0%, #fda4af 100%)', text: '#ffffff' },
  { name: '深邃星空', bg: 'linear-gradient(135deg, #3b82f6 0%, #8b5cf6 100%)', text: '#ffffff' },
  { name: '翡翠竹林', bg: 'linear-gradient(135deg, #10b981 0%, #06b6d4 100%)', text: '#ffffff' },
  { name: '烈焰骄阳', bg: 'linear-gradient(135deg, #f59e0b 0%, #ef4444 100%)', text: '#ffffff' },
  { name: '珊瑚落日', bg: 'linear-gradient(135deg, #ec4899 0%, #f43f5e 100%)', text: '#ffffff' },
  { name: '极光幻境', bg: 'linear-gradient(135deg, #14b8a6 0%, #6366f1 100%)', text: '#ffffff' }
];

// Profile interfaces
interface EmergencyContact {
  name: string;
  relationship: string;
  phone: string;
  notes: string;
}

interface Profile {
  name: string;
  age: number;
  gender: '男' | '女' | '其他' | '选择保密';
  weight: number; // stores raw value in kg
  weightUnit: 'kg' | 'lbs';
  avatarUrl: string; // custom base64 or empty for preset
  avatarPresetIndex: number;
  bio: string;
  emergencyContact: EmergencyContact;
}

// Initial default state
const initialData: Profile = {
  name: '张云飞',
  age: 28,
  gender: '男',
  weight: 72.5,
  weightUnit: 'kg',
  avatarUrl: '',
  avatarPresetIndex: 0,
  bio: '终身学习者，热爱生活、徒步与科技。保持探索，永不止步。',
  emergencyContact: {
    name: '林静宜',
    relationship: '配偶',
    phone: '138-0013-8000',
    notes: '若发生紧急情况请立即通知，血型为O型。'
  }
};

// Main states
const profile = reactive<Profile>({ ...initialData });
const isEditing = ref(false);
const showAvatarPicker = ref(false);
const fileInputRef = ref<HTMLInputElement | null>(null);

// Validation error states
const errors = reactive({
  name: '',
  age: '',
  weight: '',
  emergencyName: '',
  emergencyPhone: ''
});

// Toast system state
const showToast = ref(false);
const toastMessage = ref('');
const toastType = ref<'success' | 'error' | 'info'>('success');
let toastTimeout: NodeJS.Timeout | null = null;

const triggerToast = (msg: string, type: 'success' | 'error' | 'info' = 'success') => {
  toastMessage.value = msg;
  toastType.value = type;
  showToast.value = true;
  if (toastTimeout) clearTimeout(toastTimeout);
  toastTimeout = setTimeout(() => {
    showToast.value = false;
  }, 3500);
};

// Load saved data on mount
onMounted(() => {
  const saved = localStorage.getItem('user_profile_vue_app');
  if (saved) {
    try {
      const parsed = JSON.parse(saved);
      Object.assign(profile, parsed);
    } catch (e) {
      console.error('Failed to parse saved profile data:', e);
    }
  }
});

// Weight Computeds (converts display vs model stores in kg)
const displayWeight = computed({
  get() {
    if (profile.weightUnit === 'lbs') {
      return parseFloat((profile.weight * 2.20462).toFixed(1));
    }
    return profile.weight;
  },
  set(newValue: number) {
    if (profile.weightUnit === 'lbs') {
      profile.weight = parseFloat((newValue / 2.20462).toFixed(1));
    } else {
      profile.weight = newValue;
    }
  }
});

// Fit status calculation based on standard Weight metrics (Assuming generic height 1.75m for illustrative context)
const fitnessStatus = computed(() => {
  // Simple check
  if (profile.weight <= 0) return { label: '未设定', color: 'bg-slate-100 text-slate-700 border-slate-200' };
  
  // BMI helper with average 1.75m height for high-fidelity indicators
  const estimatedBMI = profile.weight / (1.75 * 1.75);
  
  if (estimatedBMI < 18.5) {
    return { label: '偏瘦 / Underweight', color: 'bg-amber-50 text-amber-600 border-amber-200' };
  } else if (estimatedBMI >= 18.5 && estimatedBMI < 24) {
    return { label: '健康常规 / Fit & Balanced', color: 'bg-emerald-50 text-emerald-600 border-emerald-200' };
  } else if (estimatedBMI >= 24 && estimatedBMI < 28) {
    return { label: '超重范围 / Overweight', color: 'bg-orange-50 text-orange-600 border-orange-200' };
  } else {
    return { label: '严重超重 / High BMI', color: 'bg-rose-50 text-rose-600 border-rose-200' };
  }
});

// Unit toggler that keeps correct metric values
const setWeightUnit = (unit: 'kg' | 'lbs') => {
  if (profile.weightUnit === unit) return;
  profile.weightUnit = unit;
  triggerToast(`体重单位已切换为 ${unit.toUpperCase()}`, 'info');
};

// Preset avatar selector
const selectPresetAvatar = (idx: number) => {
  profile.avatarUrl = ''; // Clear custom file
  profile.avatarPresetIndex = idx;
  triggerToast(`已应用预设头像: ${gradientPresets[idx].name}`, 'success');
};

// Custom avatar logic
const handleAvatarUpload = (event: Event) => {
  const target = event.target as HTMLInputElement;
  if (target.files && target.files[0]) {
    const file = target.files[0];
    if (file.size > 2 * 1024 * 1024) {
      triggerToast('图片大小不能超过 2MB', 'error');
      return;
    }
    const reader = new FileReader();
    reader.onload = (e) => {
      if (e.target?.result) {
        profile.avatarUrl = e.target.result as string;
        triggerToast('自定义头像上传成功！', 'success');
      }
    };
    reader.readAsDataURL(file);
  }
};

const triggerFileInput = () => {
  fileInputRef.value?.click();
};

const removeCustomAvatar = () => {
  profile.avatarUrl = '';
  profile.avatarPresetIndex = 0;
  triggerToast('已恢复为默认预设头像', 'info');
};

// Form Actions
const validateForm = (): boolean => {
  let isValid = true;
  
  // Name
  if (!profile.name.trim()) {
    errors.name = '姓名不能为空';
    isValid = false;
  } else if (profile.name.length > 20) {
    errors.name = '姓名长度不能超过 20 个字符';
    isValid = false;
  } else {
    errors.name = '';
  }

  // Age
  if (profile.age === null || profile.age === undefined) {
    errors.age = '年龄不能为空';
    isValid = false;
  } else if (profile.age < 0 || profile.age > 150) {
    errors.age = '请输入合理的年龄 (0-150)';
    isValid = false;
  } else {
    errors.age = '';
  }

  // Weight
  if (!profile.weight || profile.weight <= 0) {
    errors.weight = '请输入合理的体重数值';
    isValid = false;
  } else {
    errors.weight = '';
  }

  // Emergency Contact Name
  if (!profile.emergencyContact.name.trim()) {
    errors.emergencyName = '紧急联系人姓名不能为空';
    isValid = false;
  } else {
    errors.emergencyName = '';
  }

  // Emergency Contact Phone
  const phoneVal = profile.emergencyContact.phone.trim();
  if (!phoneVal) {
    errors.emergencyPhone = '紧急联系人电话不能为空';
    isValid = false;
  } else if (phoneVal.length < 5) {
    errors.emergencyPhone = '请输入有效的联系电话';
    isValid = false;
  } else {
    errors.emergencyPhone = '';
  }

  return isValid;
};

// Save Data
const saveProfile = () => {
  if (!validateForm()) {
    triggerToast('请检查输入表单，修复错误后再保存', 'error');
    return;
  }
  
  // Save to local storage
  localStorage.setItem('user_profile_vue_app', JSON.stringify(profile));
  isEditing.value = false;
  showAvatarPicker.value = false;
  triggerToast('资料已成功保存且同步至本地！', 'success');
};

// Reset to initialData
const resetProfile = () => {
  if (confirm('确认将资料重置为默认演示数据吗？未保存的更改将会丢失。')) {
    Object.assign(profile, initialData);
    localStorage.removeItem('user_profile_vue_app');
    isEditing.value = false;
    showAvatarPicker.value = false;
    // reset errors
    Object.keys(errors).forEach(key => {
      errors[key as keyof typeof errors] = '';
    });
    triggerToast('已恢复至系统默认演示数据', 'info');
  }
};

// Safety pocket-card rendering system
const showSafetyCard = ref(false);
</script>

    <!-- Main Container -->
    <main class="max-w-4xl mx-auto px-4 mt-6">
      
      <!-- Profile Header / Banner Card -->
      <section class="bg-white rounded-3xl shadow-sm border border-slate-100 overflow-hidden mb-6 relative">
        <!-- Abstract Soft Wave Background -->
        <div class="h-40 sm:h-48 bg-slate-900 overflow-hidden relative">

          <div class="absolute -top-16 -left-16 w-64 h-64 bg-violet-400 rounded-full blur-3xl opacity-35"></div>
          <div class="absolute -bottom-16 -right-16 w-80 h-80 bg-pink-400 rounded-full blur-3xl opacity-35"></div>
          
          <div class="absolute inset-0 flex items-center justify-between px-6 sm:px-8 z-20">

              <span class="inline-flex items-center px-2.5 py-0.5 rounded-full text-xs font-medium bg-white/10 backdrop-blur-md border border-white/10 text-white mb-2">
                <Sparkles class="w-3 h-3 mr-1 text-amber-300" />
                个人账户中心
              </span>
              <p class="text-xs text-white/50">最后同步时间: {{ new Date().toLocaleDateString() }}</p>
            </div>
            
            <button

            >
              <FileText class="w-3.5 h-3.5" />

        </div>

        <!-- Avatar & Summary Overlap Block -->
        <div class="px-6 pb-6 pt-0 relative flex flex-col sm:flex-row sm:items-end justify-between -mt-14 sm:-mt-16 z-20">
          <div class="flex flex-col sm:flex-row items-center sm:items-end text-center sm:text-left space-y-4 sm:space-y-0 sm:space-x-5">
            <!-- Avatar Container -->
            <div class="relative group">
              <!-- Preset Avatar Render -->
              <div 
                v-if="!profile.avatarUrl"
                class="w-28 h-28 sm:w-32 sm:h-32 rounded-3xl border-4 border-white shadow-md flex items-center justify-center transition duration-300"

                  {{ profile.name ? profile.name.trim().charAt(0) : 'U' }}

              <!-- Custom Uploaded Avatar Render -->
              <div

                <span 
                  class="inline-block px-2.5 py-0.5 rounded-full text-xs font-semibold"

                    'bg-sky-50 text-sky-600': profile.gender === '男',

                >
                  {{ profile.gender }}

                <span class="text-xs text-slate-450 font-medium font-mono">

              <!-- Bio Field -->
              <div v-if="!isEditing" class="mt-2 text-sm text-slate-500 max-w-md line-clamp-2">

                  maxlength="80"
                />

                </button>

        <div v-if="showSafetyCard" class="bg-gradient-to-br from-rose-500 via-rose-600 to-red-700 rounded-3xl p-6 text-white shadow-xl shadow-rose-100 mb-6 relative overflow-hidden">
          <div class="absolute top-0 right-0 -mr-6 -mt-6 w-32 h-32 bg-white/5 rounded-full blur-2xl"></div>
          
          <div class="flex items-start justify-between border-b border-white/20 pb-4">

                  <span class="text-[9px] text-rose-200/70 block">关系 / RELATIONSHIP</span>

                </div>

                  {{ profile.emergencyContact.phone }}

      <!-- Bento Grid Blocks (Profile details form / viewer) -->
